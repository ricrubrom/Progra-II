Ejercicio 8.- Escriba un programa en lenguaje Assembler que permita 
              realizar una Multiplicación de números sin signo almacenados en la memoria del microprocesador.



org 1000h
num1 dw 0003h
num2 dw 000Ah
act dw 0001h
resu dw ?
org 2000h
mov AX, num1
mov cx, num2
mult: add AX,num1
inc act
cmp act,CX
JNZ mult
mov resu,AX
hlt
end
