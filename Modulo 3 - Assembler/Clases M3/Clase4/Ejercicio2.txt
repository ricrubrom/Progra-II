ORG 1000H
PALABRA DB "HOLA MUNDO0%"
PORCENTAJE DB "%"

ORG 3000H
CONCAR: MOV AL, [BX]
        CMP AL,DL
        JNZ WHILE
        JMP FIN
WHILE:  INC CL
        INC BX
        MOV AL, [BX]
        CMP AL,DL
        JNZ WHILE
FIN:    RET


ORG 2000H
MOV BX, OFFSET PALABRA
MOV DL, PORCENTAJE
CALL CONCAR
HLT
END
