ORG 1000H
TABLA_A DB 1, 2, 5, 4, 30
TABLA_B DB 1, 7, 5, 9, 7
CONT DB ?
MENOR_A EQU 10

ORG 3000H
SUBRUTINA: PUSH AX
PUSH CX
PUSH DX
MOV CL, 1
MOV DL,[BX]
REPEAT: ADD CL,CL
INC AL
UNTIL: CMP AL,DL
JNZ REPEAT
FINR: MOV [BX],CL
POP DX
POP CX
POP AX
RET


ORG 2000H
MOV AX, OFFSET TABLA_A
MOV BX, OFFSET TABLA_B
MOV CX, OFFSET CONT
MOV DL,0
WHILE: CMP BX,CX
JZ FIN
IF: CMP BYTE PTR [BX],MENOR_A
JNS SIGUE
THEN: ADD DL,[BX]

SIGUE:PUSH AX
PUSH BX
MOV BX, AX
MOV AX, [BX]
POP BX
CMP AL,BYTE PTR [BX]
JS THEN2
JMP SIG
THEN2: CALL SUBRUTINA
SIG: POP AX
INC AX
INC BX
JMP WHILE
FIN: MOV CONT, DL
HLT
END
