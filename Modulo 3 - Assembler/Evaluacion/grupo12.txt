ORG 1000H
CADENA DB "GonzaloMateo"
TOT DB 0
VOC DB "AaEeIiOoUu"
FINVOC DB ?

ORG 3000H
SWAP: PUSH AX
PUSH BX
PUSH CX
PUSH DX
MOV CL, [BX]  ;BX:V2
MOV DX, BX    ;DX:DIREC V2
MOV BX, AX    ;PASAMOS  DIREC V1 A BX
MOV AL, [BX]  ;AX:V1
MOV [BX],CL   ;DIREC BX(V1)=V2
MOV BX, DX    ;RECUPERANDO DIREC V2
MOV [BX], AL  ;DIREC BX(V2)=V1
FINR1:POP DX
POP CX
POP BX
POP AX
RET


ES_VOCAL: PUSH CX
PUSHF
MOV DL, 0
SIGUE: CMP CL, [BX]
JZ TRUE
INC BX
DEC AL
JNZ SIGUE
JMP FINR2
TRUE: MOV DL, 0FFH
FINR2:POPF
POP CX
RET

VOCALES: PUSH CX
MOV CH, AL
PUSH BX 
CALL ES_VOCAL
CMP DL, 0FFH
JNZ SIGUEV
INC DH
SIGUEV: POP BX
MOV CL,AH
MOV AL,CH
CALL ES_VOCAL
CMP DL,0FFH
JNZ FINV
INC DH
FINV: POP CX
RET

ORG 2000H
MOV CH, OFFSET TOT - OFFSET CADENA
SUB CH,2
MOV AX, OFFSET CADENA  ;AX: DIREC V1
MOV BX, OFFSET TOT  ;BX: DIREC V2
DEC BX
MOV DH, 0 ; CONT VOC
FOR: CALL SWAP
MOV CL, [BX]
PUSH AX
MOV AH, [BX]
MOV CL, AH
POP AX
PUSH BX
MOV BX, AX
PUSH AX
MOV AH, [BX]
MOV BX, OFFSET VOC
MOV AL, OFFSET FINVOC - OFFSET VOC
CALL VOCALES
POP AX
POP BX
INC AX
DEC BX
SUB CH,2
JNS FOR
MOV TOT,DH
HLT
END
