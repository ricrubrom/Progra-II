Procedure busqueda (n:Integer;Var pos:arbol);
Begin
  If pos=Nil Then
    WriteLn ('El dato indicado no se encuentra en el arbol')
  Else
    If pos^.datos>n Then
      Begin
        pos := pos^.HI;
        busqueda (n, pos)
      End
  Else If pos^.datos<n Then
         Begin
           pos := pos^.HD;
           busqueda (n,pos)
         End;
End;
