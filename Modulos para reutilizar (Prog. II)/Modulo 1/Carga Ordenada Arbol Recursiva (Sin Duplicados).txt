Procedure cargaarbol (Var a:arbol);

Procedure insertar (Var a:arbol; n:integer);
Begin
  If a = Nil Then
    Begin
      new (a);
      a^.datos := n;
      a^.HI := Nil;
      a^.HD := Nil;
    End
  Else If a^.datos>n Then insertar (a^.HI,n)
  Else If a^.datos<n Then insertar (a^.HD,n)
End;

Var n: Integer;
Begin
  write ('Inserte numero (0 Finaliza la carga): ');
  ReadLn(n);
  If n<>0 Then
    Begin
      insertar (a,n);
      cargaarbol (a)
    End;
End;
