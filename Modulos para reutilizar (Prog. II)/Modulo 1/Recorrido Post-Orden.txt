
//Post Orden de izquierda a derecha
Procedure PostOrden(a:arbol);
Begin
  If (a<>Nil) Then
    Begin
      PostOrden(a^.HI);
      PostOrden(a^.HD);
      write (' ', a^.datos);
    End;
End;
