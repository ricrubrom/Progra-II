

//Pre Orden de izquierda a derecha
Procedure PreOrden (a:arbol);
Begin
  If a<>Nil Then
    Begin
      write (' ', a^.datos);
      PreOrden (a^.HI);
      PreOrden (a^.HD)
    End;
End;
